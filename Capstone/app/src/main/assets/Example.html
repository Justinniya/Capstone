Class Definition:

            class Person:
                def __init__(self, name, age):
                    self.name = name
                    self.age = age

                def greet(self):
                    return f"Hello, my name is {self.name} and I am {self.age} years old."

Instantiating Objects:

            person1 = Person("Alice", 30)
            person2 = Person("Bob", 25)
